\begin{recipe}{Granola}{TBD Portions}{30 Minutes}

Preheat oven to 350\degrees F.\\

\ingredient[4]{Cup}{Old-Fashioned Oats}
\ingredient[1~\fr{1}{2}]{Cup}{Nuts/Seeds}
\ingredient[\fr{3}{4}]{Tsp}{Salt}
\ingredient[\fr{1}{2}]{Tsp}{Cinnamon}
Mix dry ingredients in large mixing bowl.\\~\\~\\~\\

\ingredient[\fr{1}{2}]{Cup}{Oil}
\ingredient[\fr{1}{2}]{Cup}{Maple Syrup/Honey}
\ingredient[1]{Tsp}{Vanilla Extract}
Add wet ingredients to dry mixture and mix well.\\~\\~\\

\newstep
Transfer to baking tray lined with parchment paper.\\

\newstep
Bake 21-24 minutes, stirring half-way through.\\

\newstep
Cool completely before serving.

\end{recipe}
